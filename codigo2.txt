programa {
    // Declaração de variáveis de tipos diferentes
    inteiro contador, limite;
    real salario_base, bonus;

    // Atribuições e uso de literais
    limite = 10;
    salario_base = 2500.75;
    contador = 0;

    // Estrutura de repetição 'enquanto'
    enquanto (contador < limite) faca {
        
        // Estrutura condicional 'se' com operadores relacionais
        se (contador == 5) {
            escreva salario_base;
        } senao {
            leia bonus;
            salario_base = salario_base + bonus;
        }

        contador = contador + 1; // Operador aritmético
    }

    // Erro Léxico: O caractere '@' não pertence à linguagem
    real resultado_final;
    resultado_final = salario_base @ 1.1; 
}
